    <script>
        var Canvas = function (w, h) {
            this.width = w
            this.height = h
        }
        Canvas.prototype.init = function () {
            let oC = document.createElement('canvas')
            oC.setAttribute('width', this.width)
            oC.setAttribute('height', this.height)
            oC.setAttribute('id', 'canvas')
            oC.style.backgroundColor = '#000'
            document.body.appendChild(oC)
        }

        let canvas = new Canvas(window.innerWidth, window.innerHeight)
        canvas.init()

        let canvasCtx = document.querySelector('#canvas').getContext('2d')
        let width = document.querySelector('#canvas').getAttribute('width')
        let height = document.querySelector('#canvas').height

        function random(min, max) {
            return Math.random() * (max - min) + min
        }

        var Snow = function () { }

        Snow.prototype = {
            init() {
                this.x = random(0, width)
                this.y = 0
                this.radius = random(1, 5)
                this.velocity = random(1, 3)
                return this
            },
            draw(ctx) {
                ctx.beginPath()
                ctx.fillStyle = 'white'
                ctx.arc(this.x, this.y + this.radius, this.radius, 0, Math.PI * 2, false)
                ctx.fill()
                ctx.closePath()
                this.update()
            },
            update() {
                if (this.y + this.radius < height) {
                    this.y = this.y + this.velocity
                } else {
                    this.init()
                }
            }
        }

        let snows = []

        for (let i = 1; i <= 500; i++) {
            setTimeout(() => {
                snows.push(new Snow().init())
            }, i * 10)
        }

        (function move() {
            canvasCtx.clearRect(0, 0, width, height)
            for (let i = 0; i < snows.length; i++) {
                snows[i].draw(canvasCtx)
            }
            requestAnimationFrame(move)
        })()

    </script>
