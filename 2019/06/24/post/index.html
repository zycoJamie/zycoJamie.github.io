因为const声明的常量如果是个引用对象，那么这个常量所指向的内存中的数据，是可能被修改

的，那么如何将对象彻底冻结？借助Object.freeze

function freezeObj(obj){
  Object.freeze(obj)
  Object.keys(obj).forEach((key,idx)=>{
    if(typeof obj[key]=='object'){
      freezeObj(obj[key])
    }
  })
}
